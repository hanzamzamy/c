9    public class Siswa extends Warga {
10       public int noUrut;
11       public String peminatan, kelas, jurusan, sekolah;
12       public WaliKelas waliKelas;
13   
14       public Siswa(String namalengkap, Kelas kelas, int nourut, String peminatan) {
15           super(namalengkap, "Siswa");
16           this.noUrut = nourut;
17           this.peminatan = peminatan;
18           this.kelas = kelas.namaKelas;
19           this.waliKelas = kelas.waliKelas;
20           this.jurusan = kelas.namaJurusan;
21           this.sekolah = kelas.namaSekolah;
22           Siswa[] tmpArr = new Siswa[kelas.daftarSiswa.length + 1];
23           for (int i=0;i<tmpArr.length;i++) {
24               if (i<kelas.daftarSiswa.length){
25                   tmpArr[i] = kelas.daftarSiswa[i];
26               } else {
27                   tmpArr[i] = this;
28               }
29           }
30           kelas.daftarSiswa = tmpArr;
31           kelas.jumlahSiswa += 1;
32       }
33   
34       public void informasi(boolean deskriptif) {
35           if (deskriptif) {
36               System.out.printf(
37                       """
38                       Halo! Saya %s. %s %s dari kelas %s, urutan ke-%d, peminatan %s. Saya berasal dari jurusan %s. %s\040 
39                       adalah wali kelas saya. Salam kenal. 
40                       """, this.namaLengkap, this.jabatan, this.sekolah, this.kelas, this.noUrut, this.peminatan,
41                       this.jurusan, this.waliKelas.namaLengkap
42               );
43           } else {
44               System.out.print("\nData Siswa\n");
45               System.out.printf("Nama Lengkap\t: %s\n", this.namaLengkap);
46               System.out.printf("Jabatan\t\t\t: %s\n", this.jabatan);
47               System.out.printf("Wali Kelas\t\t: %s\n", this.waliKelas.namaLengkap);
48               System.out.printf("Nomor Urut\t\t: %d\n", this.noUrut);
49               System.out.printf("Kelas\t\t\t: %s\n", this.kelas);
50               System.out.printf("Jurusan\t\t\t: %s\n", this.jurusan);
51               System.out.printf("Peminatan\t\t: %s\n", this.peminatan);
52               System.out.printf("Sekolah\t\t\t: %s\n", this.sekolah);
53               System.out.print("\n");
54           }
55       }
