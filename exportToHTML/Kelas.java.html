7    package MyJavaOOPLib;
8    
9    public class Kelas {
10       public String namaKelas, namaJurusan, namaSekolah;
11       public Siswa[] daftarSiswa = new Siswa[0];
12       public WaliKelas waliKelas;
13       public int jumlahSiswa = 0;
14   
15       public Kelas(String namakelas, String namajurusan, String namasekolah){
16           this.namaKelas = namakelas;
17           this.namaJurusan = namajurusan;
18           this.namaSekolah = namasekolah;
19       }
20   
21       public void informasi(boolean deskriptif) {
22           if (deskriptif) {
23               System.out.printf(
24                       """
25                       %s merupakan salah satu kelas dari jurusan %s di %s. Terdiri dari %d siswa yang dibina oleh %s,\040 
26                       (selaku %s %s di %s). Berikut ini data-data dari masing-masing siswa di kelas %s 
27                       """, this.namaKelas, this.namaJurusan, this.namaSekolah, this.jumlahSiswa,
28                       this.waliKelas.namaLengkap, this.waliKelas.jabatan, this.waliKelas.mataPelajaran,
29                       this.waliKelas.sekolah, this.namaKelas
30               );
31   
32               for (int i=0; i<this.jumlahSiswa; i++) {
33                   System.out.printf(
34                       "Di urutan ke-%d adalah %s. %s dari peminatan %s.\n\n", this.daftarSiswa[i].noUrut,
35                       this.daftarSiswa[i].namaLengkap, this.daftarSiswa[i].jabatan, this.daftarSiswa[i].peminatan
36                   );
37               }
38           } else {
39               System.out.print("\nData Kelas\n");
40               System.out.printf("Kelas\t\t\t: %s\n", this.namaKelas);
41               System.out.printf("Jurusan\t\t\t: %s\n", this.namaJurusan);
42               System.out.printf("Sekolah\t\t\t: %s\n", this.namaSekolah);
43               System.out.printf("Jumlah Siswa\t: %d\n", this.jumlahSiswa);
44               System.out.printf("Wali Kelas\t\t: %s\n", this.waliKelas.namaLengkap);
45               System.out.print("\n");
46   
47               for (int i=0; i<this.jumlahSiswa; i++) {
48                   System.out.printf("Nomor Urut\t\t: %d\n", this.daftarSiswa[i].noUrut);
49                   System.out.printf("Nama Lengkap\t: %s\n", this.daftarSiswa[i].namaLengkap);
50                   System.out.printf("Peminatan\t\t: %s\n", this.daftarSiswa[i].peminatan);
51                   System.out.print("\n");
52               }
53           }
54       }
55   }
