8    class Kelas(object):
9        def __init__(self, namakelas, namajurusan, namasekolah):
10           self.namaKelas = namakelas
11           self.namaJurusan = namajurusan
12           self.namaSekolah = namasekolah
13           self.daftarSiswa = []
14           self.waliKelas = None
15           self.jumlahSiswa = 0
16   
17       def informasi(self, deskriptif=False):
18           if deskriptif:
19               print(
20                   "{0} merupakan salah satu kelas dari jurusan {1} di {2}. Terdiri dari {3} siswa yang dibina oleh {4}, "
21                   "(selaku {5} {6} di {7}). Berikut ini data-data dari masing-masing siswa di kelas {0}:\n"
22                   .format(
23                       self.namaKelas, self.namaJurusan, self.namaSekolah, self.jumlahSiswa, self.waliKelas.namaLengkap,
24                       self.waliKelas.jabatan, self.waliKelas.mataPelajaran, self.waliKelas.sekolah
25                   )
26               )
27   
28               for siswa in self.daftarSiswa:
29                   print(
30                       "Di urutan ke-{0} adalah {1}. {2} dari peminatan {3}.\n"
31                       .format(
32                           siswa.noUrut, siswa.namaLengkap, siswa.jabatan, siswa.peminatan
33                       )
34                   )
35           else:
36               print("\nData Kelas")
37               print("Kelas\t\t\t: {}".format(self.namaKelas))
38               print("Jurusan\t\t\t: {}".format(self.namaJurusan))
39               print("Sekolah\t\t\t: {}".format(self.namaSekolah))
40               print("Jumlah Siswa\t: {}".format(self.jumlahSiswa))
41               print("Wali Kelas\t\t: {}".format(self.waliKelas.namaLengkap))
42               print()
43   
44               for siswa in self.daftarSiswa:
45                   print("Nomor Urut\t\t: {}".format(siswa.noUrut))
46                   print("Nama Lengkap\t: {}".format(siswa.namaLengkap))
47                   print("Peminatan\t\t: {}".format(siswa.peminatan))
48                   print()

51   class Warga(object):
52       def __init__(self, namalengkap, jabatan):
53           self.namaLengkap = namalengkap
54           self.jabatan = jabatan
55   
56       def informasi(self, deskriptif=False):
57           if deskriptif:
58               print(
59                   "Nama saya {0}. Saya adalah seorang {1}.\n"
60                   .format(
61                       self.namaLengkap, self.jabatan
62                   )
63               )
64   
65           else:
66               print("\nData Diri")
67               print("Nama Lengkap\t: {}".format(self.namaLengkap))
68               print("Jabatan\t\t\t: {}".format(self.jabatan))
69               print()

72   class WaliKelas(Warga):
73       def __init__(self, namalengkap, kelas, matapelajaran):
74           self.mataPelajaran = matapelajaran
75           self.kelas = kelas.namaKelas
76           self.sekolah = kelas.namaSekolah
77           kelas.waliKelas = self
78           Warga.__init__(self, namalengkap, 'Guru')
79   
80       def informasi(self, deskriptif=False):
81           if deskriptif:
82               print(
83                   "Perkenalkan. Saya {0}, {1} {2} dan wali kelas dari {3} di {4}.\n"
84                   .format(
85                       self.namaLengkap, self.jabatan, self.mataPelajaran, self.kelas, self.sekolah
86                   )
87               )
88   
89           else:
90               print("\nData Wali Kelas")
91               print("Nama Lengkap\t: {}".format(self.namaLengkap))
92               print("Jabatan\t\t\t: {}".format(self.jabatan))
93               print("Mata Pelajaran\t: {}".format(self.mataPelajaran))
94               print("Kelas\t\t\t: {}".format(self.kelas))
95               print("Sekolah\t\t\t: {}".format(self.sekolah))
96               print()

99   class Siswa(Warga):
100      def __init__(self, namalengkap, kelas, nourut, peminatan):
101          self.noUrut = nourut
102          self.peminatan = peminatan
103          Warga.__init__(self, namalengkap, 'Siswa')
104          self.kelas = kelas.namaKelas
105          self.waliKelas = kelas.waliKelas
106          self.jurusan = kelas.namaJurusan
107          self.sekolah = kelas.namaSekolah
108          kelas.jumlahSiswa += 1
109          kelas.daftarSiswa.append(self)
110  
111      def informasi(self, deskriptif=False):
112          if deskriptif:
113              print(
114                  "Halo! Saya {0}. {1} {2} dari kelas {3}, urutan ke-{4}, peminatan {5}. Saya berasal dari "
115                  "jurusan {6}. {7} adalah wali kelas saya. Salam kenal.\n"
116                  .format(
117                      self.namaLengkap, self.jabatan, self.sekolah, self.kelas, self.noUrut, self.peminatan, self.jurusan,
118                      self.waliKelas.namaLengkap
119                  )
120              )
121  
122          else:
123              print("\nData Siswa")
124              print("Nama Lengkap\t: {}".format(self.namaLengkap))
125              print("Jabatan\t\t\t: {}".format(self.jabatan))
126              print("Wali Kelas\t\t: {}".format(self.waliKelas.namaLengkap))
127              print("Nomor Urut\t\t: {}".format(self.noUrut))
128              print("Kelas\t\t\t: {}".format(self.kelas))
129              print("Jurusan\t\t\t: {}".format(self.jurusan))
130              print("Peminatan\t\t: {}".format(self.peminatan))
131              print("Sekolah\t\t\t: {}".format(self.sekolah))
132              print()
