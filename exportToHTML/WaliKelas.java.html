9    public class WaliKelas extends Warga {
10       public String mataPelajaran, kelas, sekolah;
11   
12       public WaliKelas(String namalengkap, Kelas kelas, String matapelajaran) {
13           super(namalengkap, "Guru");
14           this.mataPelajaran = matapelajaran;
15           this.kelas = kelas.namaKelas;
16           this.sekolah = kelas.namaSekolah;
17           kelas.waliKelas = this;
18       }
19   
20       public void informasi(boolean deskriptif) {
21           if (deskriptif) {
22               System.out.printf(
23                       "Perkenalkan. Saya %s, %s %s dan wali kelas dari %s di %s.\n\n", this.namaLengkap, this.jabatan,
24                       this.mataPelajaran, this.kelas, this.sekolah
25               );
26           } else {
27               System.out.print("\nData Wali Kelas\n");
28               System.out.printf("Nama Lengkap\t: %s\n", this.namaLengkap);
29               System.out.printf("Jabatan\t\t\t: %s\n", this.jabatan);
30               System.out.printf("Mata Pelajaran\t: %s\n", this.mataPelajaran);
31               System.out.printf("Kelas\t\t\t: %s\n", this.kelas);
32               System.out.printf("Sekolah\t\t\t: %s\n", this.sekolah);
33               System.out.print("\n");
34           }
35       }
36   }
