4    /* 
5     * DESKRIPSI: 
6     * Fungsi untuk program 3.1. 
7     * Mengumpulkan input dari user dengan mencetak pertanyaan/perintah, kemudian meminta user 
8     * untuk memberi input, dan menyimpan input ke variabel. Input akan dicetak ulang. 
9     * Fungsi yang digunakan: printf() dan scanf(). 
10    * 
11    * PARAMETER: 
12    * Tipe Nama    Keterangan 
13    * void -       Fungsi ini tidak memerlukan parameter. 
14    * 
15    * NILAI KEMBALI: 
16    * Tipe Nama    Keterangan 
17    * int  -       Fungsi mengembalikan kode keluar berupa angka 0 jika berhasil. 
18    * 
19    * KESALAHAN: 
20    * 1    Memasukan nilai melebihi atau kurang dari nilai maksimum dari variabel. 
21    *      Contoh: panjang string, nilai maksimum tipe data int, dll. 
22    * 2    Memasukan nilai non bilangan bulat ke variabel angka (int). 
23    *      Menyebabkan input terbagi menjadi dua. Contoh: input = 3.14, 
24    *      nilai variabel angka = 3, nilai variabel nama = .14 (terisi otomatis). 
25    * 3    Memasukan karakter non angka ke variabel bilangan (int, float, double, long, dll.). 
26    *      Menyebabkan input terbaca sebagai 0 dan karakter masuk ke variabel selanjutnya. 
27    * 
28    * GALAT: 
29    * 1    Karakter spasi tidak diizinkan. Karakter spasi berperilaku setara dengan ENTER. 
30    *      Menyebabkan input yang mengandung spasi masuk ke variabel yang berbeda (terpecah). 
31    * 2    Residu data input otomatis menjadi nilai variabel selanjutnya (periksa KESALAHAN 2, 
32    *      3, dan GALAT 1). 
33    * 3    Kesalahan pada format scanf() untuk variabel pecahan (double). Kode sumber dari modul 
34    *      menggunakan %d untuk menampilkan nilai desimal dari input (dengan tipe data integer). 
35    */
36   int program3d1(void){
37       char karakter, nama[50], nis[50];
38       int angka;
39       double pecahan;
40   
41       printf("Masukan satu karakter : ");
42       scanf("%c", &karakter );
43       printf("Masukan bilangan bulat : ");
44       scanf("%i", &angka );
45       printf("Masukan nama Anda (maks. 20 karakter) : ");
46       scanf("%s", &nama );
47       printf("Masukan NIS Anda (maks. 10 karakter) :");
48       scanf("%s", &nis );
49       printf("Masukan bilangan pecahan :");
50       scanf("%d", &pecahan );
51   
52       printf("\nKarakter = %c\n", karakter );
53       printf("Bilangan bulat = %i\n", angka);
54       printf("Nama = %s\n", nama);
55       printf("NIS = %s\n", nis);
56       printf("Bilangan pecahan = %d\n", pecahan);
57       return 0;
58   }

60   /* 
61    * DESKRIPSI: 
62    * Fungsi tambahan untuk program 3.1 (revisi). 
63    * Membersihkan buffer pada stream stdin. Menghapus data residual dengan cara mengambil nilai 
64    * dari buffer satu per satu (menggunakan pengulangan while) hingga memenuhi kondisional 
65    * di mana nilai = \n dan berada di akhir. Fungsi yang digunakan: getchar(). 
66    * 
67    * PARAMETER: 
68    * Tipe Nama    Keterangan 
69    * void -       Fungsi ini tidak memerlukan parameter. 
70    * 
71    * NILAI KEMBALI: 
72    * Tipe Nama    Keterangan 
73    * void -       Fungsi tidak memberikan nilai kembali. 
74    */
75   void bersihkanBuffer(void){
76       int buffer = getchar();
77   
78       while (buffer != '\n' && buffer != EOF){
79           buffer = getchar();
80       }
81   }

89    * 
90    * PARAMETER: 
91    * Tipe Nama    Keterangan 
92    * void -       Fungsi ini tidak memerlukan parameter. 
93    * 
94    * NILAI KEMBALI:

99    * 1    Memasukan nilai melebihi atau kurang dari nilai maksimum dari variabel. 
100   *      Contoh: panjang string, nilai maksimum tipe data int, dll. 
101   * 2    Memasukan karakter non angka ke variabel bilangan (int, float, double, long, dll.). 
102   *      Menyebabkan input terbaca sebagai 0. 
103   * 
104   * PENYELESAIAN: 
105   * 1    Mengatur ulang set karakter yang diterima oleh scanf(). Menggunakan bitwise XOR 
106   *      sehingga saat karakter yang dimasukan bukan ENTER (\n) logika = TRUE dan pembacaan 
107   *      stdin berlanjut hingga karakter yang dimasukan sama dengan ENTER (logika = FALSE, 
108   *      pembacaan selesai). AWAKUTU GALAT 1. 
109   * 2    Membersihkan data temporal stdin sebelum membaca data input (periksa fungsi 
110   *      bersihkanBuffer()). AWAKUTU GALAT 2. 
111   * 3    Mengganti format ke %lf karena tipe data variabel pecahan adalah double. 
112   *      AWAKUTU GALAT 3. 
113   */
114  int program3d1rev(void){
115      char karakter, nama[50], nis[50];
116      int angka;
117      double pecahan;
118  
119      printf("Masukan satu karakter : ");
120      scanf("%c", &karakter);
121      bersihkanBuffer();
122      printf("Masukan bilangan bulat : ");
123      scanf("%i", &angka);
124      bersihkanBuffer();
125      printf("Masukan nama Anda (maks. 20 karakter) : ");
126      scanf("%[^\n]s", nama);
127      bersihkanBuffer();
128      printf("Masukan NIS Anda (maks. 10 karakter) :");
129      scanf("%[^\n]s", nis);
130      bersihkanBuffer();
131      printf("Masukan bilangan pecahan :");
132      scanf("%lf", &pecahan);
133      bersihkanBuffer();
134  
135      printf("\nKarakter = %c\n", karakter);
136      printf("Bilangan bulat = %i\n", angka);
137      printf("Nama = %s\n", nama);
138      printf("NIS = %s\n", nis);
139      printf("Bilangan pecahan = %lf\n", pecahan);
140      return 0;
141  }

200      double p,l,luas;
201  
202      printf("Masukan panjang : ");
203      scanf("%lf", &p );
204      printf("Masukan lebar : ");
205      scanf("%lf", &l );
206      luas = p * l;
207  
208      printf("\nLuas = %lf\n", luas);
